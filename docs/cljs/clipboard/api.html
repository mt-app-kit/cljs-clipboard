(defn copy-text!
  [text]
  (letfn [(f0 [] (-> "text-clipboard" dom/get-element-by-id .-value js/navigator.clipboard.writeText))]
         (temporary-components/append-component! [views/text-clipboard text] f0)
         (temporary-components/remove-component!)))
